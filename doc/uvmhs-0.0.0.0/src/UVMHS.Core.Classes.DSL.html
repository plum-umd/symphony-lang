class Arrow a where (⇨) ∷ a → a → a
class Apply a where (⊙) ∷ a → a → a
class Tup a where tup ∷ (ToIter a t) ⇒ t → a

(⊙$) ∷ (Apply e) ⇒ e → e → e
(⊙$) = (⊙)

(⊙⋆) ∷ (Apply e,ToIter e t) ⇒ e → t → e
(⊙⋆) x = fold𝐼 x (flip (⊙)) ∘ iter
